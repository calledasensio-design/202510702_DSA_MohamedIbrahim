

#include <iostream>
#include <vector>

int main() {
    // 1. Create a variable to store the chosen number
    int chosenNumber = 7;
    
    // 2. Create a variable to store the user's input
    int userGuess = 0; 
    
    // 3. Create an empty array/list to keep all the numbers entered by the user
    std::vector<int> attempts;

    // 4. Use a while loop that continues as long as the guess is not equal to the chosen number
    while (userGuess != chosenNumber) {
        
        // 5. Ask the user to enter a number and store the value
        std::cout << "Enter a number: ";
        std::cin >> userGuess;

        // 6. Immediately save the value into the array/list
        attempts.push_back(userGuess);

        // 7. Use if/else statements to compare the user's guess with the chosen number
        if (userGuess > chosenNumber) {
            std::cout << "Too high\n";
        } else if (userGuess < chosenNumber) {
            std::cout << "Too low\n";
        } else {
            std::cout << "Correct\n";
        }
    }

    // 8. After the loop finishes, use a for loop to display all stored guesses
    std::cout << "\nYour guesses were: ";
    for (int i = 0; i < attempts.size(); i++) {
        std::cout << attempts[i] << " ";
    }
    std::cout << "\n";

    return 0;
}
